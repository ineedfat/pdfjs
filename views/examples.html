var doc = new pdfJS.doc('letter', 'portrait'); doc.addPage(); doc.currentPage.graphic('translate', 160, 400); doc.currentPage.graphic('rect', 50, 50, 100, 100); doc.currentPage.graphic('grayFill', 1); doc.currentPage.graphic('grayStroke', 0.5); doc.currentPage.graphic('paintPath', 'B'); doc.currentPage.graphic('grayFill', 0); doc.currentPage.graphic('rect', 75, 75, 100, 100); doc.currentPage.graphic('grayFill', 0); doc.currentPage.graphic('grayStroke', 0.5); doc.currentPage.graphic('paintPath', 'B'); doc.currentPage.graphic('grayFill', 0); doc.currentPage.text('beginText'); doc.currentPage.text('fontStyle', null, null, 22); doc.currentPage.text('showText', 'Pretty Cool Huh?'); doc.currentPage.text('endText');

Note

Here shows a basic rect operator that draws a rectangular path by specifying the x-coord, y-coord, width, and height. Drawing the path alone will not actually render anything on the page. We must first specify the stroke and fill color and then finally close the path by using the paintPath operator. Also note that in this example, we are using the grayscale color space. Generally all color value ranges from 0.0 to 1.0 unless otherwise specified.

var doc = new pdfJS.doc('letter', 'portrait'); var drawRect = function() { doc.currentPage.graphic('rect', 0, 0, 100, 100); doc.currentPage.graphic('grayStroke', .5); doc.currentPage.graphic('paintPath', 'B'); doc.currentPage.graphic('grayFill', 0); doc.currentPage.graphic('popState'); }; doc.addPage(); doc.currentPage.graphic('translate', 160, 400); doc.currentPage.graphic('pushState'); doc.currentPage.graphic('grayFill', 1); drawRect(); doc.currentPage.graphic('translate', 100, 100); doc.currentPage.graphic('grayFill', 0.4); drawRect(); doc.currentPage.graphic('translate', 100, -100); doc.currentPage.graphic('grayFill', 0.2); drawRect(); doc.currentPage.graphic('translate', 10, -50); doc.currentPage.text('beginText'); doc.currentPage.text('fontStyle', null, null, 22); doc.currentPage.text('showText', 'Push and Pop :}'); doc.currentPage.text('endText');

Note

Note the amount of code required to generate a simple object like the filled square in the above example. This example explores how we can group repeated code and reduces our code footprint. Also PDF (as well as PostScript) allows graphic state to get pushed (save) on to the stack and pop (restore) off the stack to recover to the original configuration.

var doc = new pdfJS.doc('letter', 'portrait'); doc.addPage(); var imgXObj = doc.newImage('./img/sample.jpg', true); imgXObj.addImageToPage(doc.currentPage, 60, 300, 500, 200); doc.currentPage.graphic('translate', 400, 475); doc.currentPage.text('beginText'); doc.currentPage.text('fontStyle', null, null, 15); doc.currentPage.text('showText', 'Miyajima, Japan'); doc.currentPage.text('endText');

Note

pdfJS uses HTML5 canvas internally to support different kinds of image format. The newImage function accepts a string url to an image, HTMLImageElement, or HTMLCavasElement as graphics to render.
